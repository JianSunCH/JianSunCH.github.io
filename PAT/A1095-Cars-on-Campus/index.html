1095 Cars on Campus

Zhejiang University has 8 campuses and a lot of gates. From each gate we can collect the in/out times and the plate numbers of the cars crossing the gate. Now with all the information available, you are supposed to tell, at any specific time point, the number of cars parking on campus, and at the end of the day find the cars that have parked for the longest time period.

Input Specification:

Each input file contains one test case. Each case starts with two positive integers N (≤10​4​​), the number of records, and K (≤8×10​4​​) the number of queries. Then N lines follow, each gives a record in the format:

plate_number hh:mm:ss status

where plate_number is a string of 7 English capital letters or 1-digit numbers; hh:mm:ss represents the time point in a day by hour:minute:second, with the earliest time being 00:00:00 and the latest 23:59:59; and status is either in or out.

Note that all times will be within a single day. Each in record is paired with the chronologically next record for the same car provided it is an out record. Any in records that are not paired with an out record are ignored, as are out records not paired with an in record. It is guaranteed that at least one car is well paired in the input, and no car is both in and out at the same moment. Times are recorded using a 24-hour clock.

Then K lines of queries follow, each gives a time point in the format hh:mm:ss. Note: the queries are given in ascending order of the times.

Output Specification:

For each query, output in a line the total number of cars parking on campus. The last line of output is supposed to give the plate number of the car that has parked for the longest time period, and the corresponding time length. If such a car is not unique, then output all of their plate numbers in a line in alphabetical order, separated by a space.

解答

#include <bits/stdc++.h>
using namespace std;
const int max_n = 10001;

struct Car
{
    char id[8];
    int time; //以s为单位
    char status[4];
} all[max_n], valid[max_n];

bool cmpByIdAndTime(Car c1, Car c2)
{
    if (strcmp(c1.id, c2.id))
        return strcmp(c1.id, c2.id) < 0;
    else
        return c1.time < c2.time;
}

bool cmpByTime(Car c1, Car c2)
{
    return c1.time < c2.time;
}

int timeToInt(int hh, int mm, int ss)
{
    return hh * 3600 + mm * 60 + ss;
}

int num = 0;               //有效的记录数
map<string, int> parkTime; //key:车牌号,value:总停留时间

int main()
{
    int N, K;
    scanf("%d%d", &N, &K);
    int hh, mm, ss;
    for (int i = 0; i < N; i++)
    {
        scanf("%s %d:%d:%d %s", all[i].id, &hh, &mm, &ss, all[i].status);
        all[i].time = timeToInt(hh, mm, ss);
    }
    sort(all, all + N, cmpByIdAndTime); //按车牌号和时间排序

    int max_time = -1;                //记录最长的停留时间
    for (int i = 0; i < N - 1; i++) //遍历所有记录
    {
        if (!strcmp(all[i].id, all[i + 1].id) && !strcmp(all[i].status, "in") && !strcmp(all[i + 1].status, "out")) //是同一辆车, i为in, i + 1为out
        {
            valid[num++] = all[i]; //配对的,存入valid数组
            valid[num++] = all[i + 1];
            int park_time = all[i + 1].time - all[i].time; //这辆车停车的时间
            if (parkTime.count(all[i].id) == 0)               //map中还没有这辆车
                parkTime[all[i].id] = 0;
            parkTime[all[i].id] += park_time;               //否则增加停车时间
            max_time = max(max_time, parkTime[all[i].id]); //更新最大停车时间
        }
    }
    sort(valid, valid + num, cmpByTime); //把有效记录按时间从小到大排序

    int now = 0, numCar = 0; //now为不超过当前查询时间的记录,numCae为当前校园内的车辆
    for (int i = 0; i < K; i++)
    {
        scanf("%d:%d:%d", &hh, &mm, &ss);
        int temp_time = timeToInt(hh, mm, ss);
        //now处理至当前查询时间
        while (now < num && valid[now].time <= temp_time)
        {
            if (!strcmp(valid[now].status, "in"))
                numCar++; //车辆进入
            else
                numCar--;
            now++; //指向下一条记录
        }
        printf("%d\n", numCar); //输出该时刻校园内车辆数
    }

    map<string, int>::iterator it; //遍历所有车牌号
    for (it = parkTime.begin(); it != parkTime.end(); it++)
        if (it->second == max_time) //输出最长停留时间的车牌号
            printf("%s ", it->first.c_str());

    //最长的停留时间
    printf("%02d:%02d:%02d\n", max_time / 3600, max_time % 3600 / 60, max_time % 60);
    return 0;
}
