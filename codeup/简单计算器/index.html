问题 A: 简单计算器

时间限制: 1 Sec  内存限制: 32 MB

题目描述

读入一个只包含 +, -, *, / 的非负整数计算表达式，计算该表达式的值。

输入

测试输入包含若干测试用例，每个测试用例占一行，每行不超过200个字符，整数和运算符之间用一个空格分隔。没有非法表达式。当一行中只有0时输入结束，相应的结果不要输出。

输出

对每个测试用例输出1行，即该表达式的值，精确到小数点后2位。

样例输入

30 / 90 - 26 + 97 - 5 - 6 - 13 / 88 * 6 + 51 / 29 + 79 * 87 + 57 * 92 
0

样例输出

12178.21

解答

#include 
using namespace std;

struct node
{
	double num;
	char op;
	bool flag; //true为操作数,false为操作符
};

string str;
stack st;	//操作符栈
queue q;	 //后缀表达式队列
map mp; //定义操作符的优先级

void infix_to_postfix() //中缀表达式转后缀表达式
{
	double num;
	node temp;
	for (int i = 0; i < str.length();)
	{
		if (str[i] >= '0' && str[i] <= '9') 数字 { temp.flag="true;" temp.num="str[i++]" - '0'; 记录这个数的第一位 while (i < str.length() && str[i]>= '0' && str[i] <= 10 '9') { temp.num="temp.num" * + (str[i] - '0'); 更新这个数 i++; } q.push(temp); else 操作符 temp.flag="false;" 栈顶元素的操作符优先级高于该操作符,pop to q while (!st.empty() && mp[str[i]] <="mp[st.top().op])" q.push(st.top()); st.pop(); temp.op="str[i];" st.push(temp); 操作符栈还有操作符,pop (!st.empty()) double calc() 计算后缀表达式 tmp1, tmp2; node cur, temp; (!q.empty()) cur="q.front();" q.pop(); if (cur.flag) st.push(cur); 为操作数,直接压入栈 tmp2="st.top().num;" 第2操作数 tmp1="st.top().num;" 第1操作数 (cur.op="=" '+') '-') '*') 计算后的操作数压入栈 return st.top().num; int main() mp['+']="mp['-']" = 1; 操作数优先级 mp['*']="mp['/']" 2; (getline(cin, str), str !="0" ) for (string::iterator it="str.end();" it--) (*it="=" ' ') str.erase(it); infix_to_postfix(); printf("%.2f\n", calc()); 0;
